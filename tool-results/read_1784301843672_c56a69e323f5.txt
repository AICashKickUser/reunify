     4→function daysAgo(days: number): Date {
     5→  const d = new Date()
     6→  d.setDate(d.getDate() - days)
     7→  d.setHours(10, 0, 0, 0)
     8→  return d
     9→}

    11→function daysFromNow(days: number): Date {
    12→  const d = new Date()
    13→  d.setDate(d.getDate() + days)
    14→  d.setHours(10, 0, 0, 0)
    15→  return d
    16→}
